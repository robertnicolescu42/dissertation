  public messages: any[] = [];

  constructor(private alertService: AlertService) {
    this.alertService.connect().subscribe((message) => {
      this.messages.push(message);

      if (message.plantIndex === this.plantIndex) {
        this.alertGroups.push(message);
      }
    });
  }
